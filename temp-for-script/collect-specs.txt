#!/bin/bash
###############################################################################
# collect-pi-withnail-specs.sh
#
# Self-contained, read-only Bash script to gather comprehensive technical
# specifications and current state of a Raspberry Pi 5 running Hermes Agent
# (Nous Research) + Withnail (autonomous CEO bot) + Ollama.
#
# Designed for:
#   - gt-hermes user on hermes-pi hostname
#   - Raspberry Pi OS Lite (Bookworm)
#   - Pi 5 (8GB RAM class) + 128GB SD card
#   - Local-first / air-gapped lean setups with Tailscale/SSH
#
# Output: Timestamped Org-mode file + LATEST symlink in ~/pi-specs/
#         Optimized for scp retrieval and ingestion into org-roam / Tsuioku
#         knowledge base (laptop + GitHub gptix/forgrok).
#
# Usage:
#   ./collect-pi-withnail-specs.sh [--help]
#
# After run: scp the .org file(s) back to laptop for processing/ingestion.
# Re-run before and after any significant configuration changes to maintain
# living baselines and history in the knowledge base.
#
# Script version: 2026-06-04-v1.0
# Author: Generated for Joji Teira (@joji_teira) per detailed specification.
###############################################################################

set -uo pipefail

# -----------------------------------------------------------------------------
# Configuration & Paths (tailored but portable)
# -----------------------------------------------------------------------------
SCRIPT_VERSION="2026-06-04-v1.0"
OUTPUT_DIR="${HOME}/pi-specs"
DATE_STAMP=$(date +%Y%m%d-%H%M)
OUTPUT_FILE="${OUTPUT_DIR}/pi-withnail-specs-hermes-pi-${DATE_STAMP}.org"
LATEST_LINK="${OUTPUT_DIR}/pi-withnail-specs-latest.org"

# Known environment hints (used for graceful tailoring; script remains portable)
KNOWN_USER="gt-hermes"
KNOWN_HOST="hermes-pi"
KNOWN_HERMES_DIR="${HOME}/.hermes"
KNOWN_OLLAMA_DEFAULTS=("${HOME}/.ollama" "/usr/share/ollama/.ollama" "/var/lib/ollama")

# -----------------------------------------------------------------------------
# Helper Functions
# -----------------------------------------------------------------------------
print_usage() {
    cat << 'EOF'
collect-pi-withnail-specs.sh - Raspberry Pi + Hermes/Withnail/Ollama spec collector

USAGE:
  ./collect-pi-withnail-specs.sh [--help]

DESCRIPTION:
  Gathers detailed hardware, OS, networking, Ollama, Hermes Agent, Withnail,
  services, runtime state, and key configs. Outputs a structured Org-mode file
  suitable for long-term knowledge-base tracking and configuration improvement.

  The script is read-only (except for creating the output file and symlink).
  It uses command existence checks and continues gracefully on missing tools.

  Recommended: Run before/after major changes (agent upgrades, config edits,
  hardware tweaks) so the knowledge base contains clear before/after baselines.

OPTIONS:
  --help    Show this help message and exit.

OUTPUT:
  ${HOME}/pi-specs/pi-withnail-specs-hermes-pi-YYYYMMDD-HHMM.org
  ${HOME}/pi-specs/pi-withnail-specs-latest.org (symlink)

After collection, use scp from your laptop to retrieve the file(s), then
ingest/process into your org-roam / Tsuioku system.
EOF
}

log_to_file() {
    printf '%s\n' "$1" >> "$OUTPUT_FILE"
}

section() {
    local level="$1"
    local title="$2"
    log_to_file ""
    if [[ "$level" == "1" ]]; then
        log_to_file "* ${title}"
    else
        log_to_file "** ${title}"
    fi
}

subsection() {
    log_to_file "*** $1"
}

note() {
    log_to_file "NOTE: $1"
}

property() {
    log_to_file ":$1: $2"
}

command_exists() {
    command -v "$1" >/dev/null 2>&1
}

sanitize_yaml() {
    local file="$1"
    if [[ ! -f "$file" ]]; then
        echo "File not found: $file"
        return
    fi
    sed -E '
        s/([a-zA-Z0-9_\-]{24,})/[REDACTED_TOKEN]/g;
        s/(api_key|token|secret|password|private_key|bearer|auth)[[:space:]]*:[[:space:]]*.*/\1: [REDACTED]/I;
    ' "$file"
}

# -----------------------------------------------------------------------------
# Collection Functions (modular)
# -----------------------------------------------------------------------------
collect_metadata() {
    section 1 "Collection Metadata & Quick Facts"

    log_to_file "#+TITLE: Raspberry Pi + Withnail/Hermes/Ollama Specifications"
    log_to_file "#+DATE: $(date '+%Y-%m-%d %H:%M:%S %Z')"
    log_to_file "#+AUTHOR: collect-pi-withnail-specs.sh ${SCRIPT_VERSION}"
    log_to_file "#+PURPOSE: Baseline for configuration improvement and long-term knowledge-base tracking (org-roam / Tsuioku + GitHub)"
    log_to_file ""

    log_to_file ":PROPERTIES:"
    property "SCRIPT_VERSION" "${SCRIPT_VERSION}"
    property "COLLECTED_LOCAL" "$(date '+%Y-%m-%d %H:%M:%S %Z')"
    property "COLLECTED_UTC" "$(date -u '+%Y-%m-%d %H:%M:%S UTC')"
    property "HOSTNAME" "$(hostname 2>/dev/null || echo 'unknown')"
    property "USERNAME" "$(whoami 2>/dev/null || echo 'unknown')"
    property "OUTPUT_FILE" "${OUTPUT_FILE}"
    property "PI_MODEL" "To be filled from hardware section"
    property "RAM_GB" "To be filled"
    property "HERMES_VERSION" "To be filled"
    property "OLLAMA_VERSION" "To be filled"
    property "WITHNAIL_DETECTED" "To be filled"
    log_to_file ":END:"
    log_to_file ""
    log_to_file "*Purpose*: This file captures a point-in-time snapshot of the Raspberry Pi hardware, OS, and the full autonomous agent stack (Ollama + Hermes Agent + Withnail CEO bot). Use it for troubleshooting, upgrade planning, resource tuning, and maintaining a living history in the knowledge base. Re-run before and after changes."
}

collect_hardware() {
    section 1 "Hardware Inventory & Pi-specific Details"

    subsection "SoC / Processor"
    if command_exists lscpu; then
        log_to_file "#+begin_src sh :exports none"
        lscpu 2>&1 | head -30 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    else
        note "lscpu not found"
    fi

    if [[ -f /proc/cpuinfo ]]; then
        log_to_file ""
        log_to_file "#+begin_src text"
        head -40 /proc/cpuinfo >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
        local rev
        rev=$(grep -m1 '^Revision' /proc/cpuinfo | awk '{print $3}' || true)
        if [[ -n "$rev" ]]; then
            log_to_file ""
            log_to_file "# Note: CPU revision code = ${rev} (see elinux.org/RPi_HardwareHistory for mapping)"
        fi
    fi

    subsection "Pi-specific via vcgencmd (if available)"
    if command_exists vcgencmd; then
        log_to_file "#+begin_src sh"
        echo "=== vcgencmd get_config int (key settings) ===" >> "$OUTPUT_FILE"
        vcgencmd get_config int 2>&1 | head -20 >> "$OUTPUT_FILE"
        echo "" >> "$OUTPUT_FILE"
        echo "=== Memory split ===" >> "$OUTPUT_FILE"
        vcgencmd get_mem arm 2>&1 >> "$OUTPUT_FILE"
        vcgencmd get_mem gpu 2>&1 >> "$OUTPUT_FILE"
        echo "" >> "$OUTPUT_FILE"
        echo "=== Temperature ===" >> "$OUTPUT_FILE"
        vcgencmd measure_temp 2>&1 >> "$OUTPUT_FILE"
        echo "" >> "$OUTPUT_FILE"
        echo "=== Throttled status ===" >> "$OUTPUT_FILE"
        vcgencmd get_throttled 2>&1 >> "$OUTPUT_FILE"
        echo "" >> "$OUTPUT_FILE"
        echo "=== Voltage (core) ===" >> "$OUTPUT_FILE"
        vcgencmd measure_volts core 2>&1 >> "$OUTPUT_FILE"
        echo "" >> "$OUTPUT_FILE"
        echo "=== Bootloader / Firmware ===" >> "$OUTPUT_FILE"
        vcgencmd bootloader_version 2>&1 >> "$OUTPUT_FILE"
        vcgencmd version 2>&1 | head -5 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    else
        note "vcgencmd not found (install raspberrypi-utils if needed)"
    fi

    subsection "Memory (general)"
    if command_exists free; then
        log_to_file "#+begin_src sh"
        free -h 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi
    if [[ -f /proc/meminfo ]]; then
        log_to_file "#+begin_src text"
        head -15 /proc/meminfo >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    subsection "Storage & SD Card"
    if command_exists lsblk; then
        log_to_file "#+begin_src sh"
        lsblk -f -o NAME,SIZE,TYPE,FSTYPE,MOUNTPOINT,UUID 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi
    if command_exists df; then
        log_to_file "#+begin_src sh"
        df -h 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    for dev in /sys/block/mmcblk*; do
        if [[ -d "$dev" ]]; then
            subsection "SD Card Device Info: $(basename "$dev")"
            for f in name manfid oemid serial hwrev fwrev date type; do
                if [[ -f "$dev/device/$f" ]]; then
                    printf '%s: %s\n' "$f" "$(cat "$dev/device/$f" 2>/dev/null || echo 'N/A')" >> "$OUTPUT_FILE"
                fi
            done
        fi
    done

    subsection "USB Devices"
    if command_exists lsusb; then
        log_to_file "#+begin_src sh"
        lsusb 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    else
        note "lsusb not found"
    fi

    subsection "Boot Configuration"
    for cfg in /boot/firmware/config.txt /boot/config.txt; do
        if [[ -f "$cfg" ]]; then
            log_to_file ""
            log_to_file "#+begin_src conf :exports none"
            cat "$cfg" >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
            log_to_file "Path: $cfg"
            break
        fi
    done
}

collect_os_kernel() {
    section 1 "Operating System & Kernel"

    if [[ -f /etc/os-release ]]; then
        log_to_file "#+begin_src conf"
        cat /etc/os-release >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    if command_exists lsb_release; then
        log_to_file "#+begin_src sh"
        lsb_release -a 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    log_to_file ""
    if command_exists uname; then
        log_to_file "#+begin_src sh"
        uname -a >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    if [[ -f /proc/cmdline ]]; then
        subsection "Kernel Command Line"
        log_to_file "#+begin_src text"
        cat /proc/cmdline >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    subsection "Raspberry Pi Specific Packages (installed)"
    if command_exists dpkg; then
        log_to_file "#+begin_src sh"
        dpkg -l | grep -iE 'raspberry|vcgencmd|raspberrypi' | head -20 >> "$OUTPUT_FILE" || echo "No matching packages" >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi
}

collect_networking() {
    section 1 "Networking & Connectivity"

    subsection "Network Interfaces & Addresses"
    if command_exists ip; then
        log_to_file "#+begin_src sh"
        ip -brief addr 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    elif command_exists ifconfig; then
        ifconfig 2>&1 | head -30 >> "$OUTPUT_FILE"
    fi

    subsection "Hostname & DNS"
    log_to_file "Hostname: $(hostname 2>/dev/null || echo 'unknown')"
    if command_exists hostnamectl; then
        hostnamectl 2>&1 | head -10 >> "$OUTPUT_FILE"
    fi

    subsection "Tailscale Status (if present)"
    if command_exists tailscale; then
        log_to_file "#+begin_src sh"
        tailscale status 2>&1 | head -15 >> "$OUTPUT_FILE" || true
        log_to_file "#+end_src"
        note "Tailscale detected — review output for connection state"
    else
        note "tailscale command not found in PATH"
    fi

    subsection "WiFi / Regulatory (if applicable, no passwords shown)"
    if command_exists iw; then
        iw reg get 2>&1 | head -5 >> "$OUTPUT_FILE" || true
    fi
    if command_exists nmcli; then
        nmcli -t -f active,ssid dev wifi 2>/dev/null | head -5 >> "$OUTPUT_FILE" || true
    fi

    subsection "Listening Services (agent-relevant ports)"
    if command_exists ss; then
        log_to_file "#+begin_src sh"
        ss -tuln 2>&1 | grep -E 'LISTEN|11434|8080|3000|5000|8000' | head -15 >> "$OUTPUT_FILE" || echo "No matching listeners" >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi
}

collect_ollama() {
    section 1 "AI Software Stack - Ollama"

    subsection "Ollama Version & Status"
    if command_exists ollama; then
        log_to_file "#+begin_src sh"
        ollama --version 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"

        subsection "Installed Models"
        log_to_file "#+begin_src sh"
        ollama ls 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"

        subsection "Currently Loaded / Running Models"
        log_to_file "#+begin_src sh"
        ollama ps 2>&1 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    else
        note "ollama command not found in PATH"
    fi

    subsection "Ollama Model Storage & Disk Usage"
    local model_dir=""
    for d in "${KNOWN_OLLAMA_DEFAULTS[@]}"; do
        if [[ -d "$d/models" ]]; then
            model_dir="$d/models"
            break
        fi
    done
    if [[ -z "$model_dir" ]]; then
        if command_exists env; then
            model_dir=$(env | grep -i OLLAMA_MODELS | head -1 | cut -d= -f2 || true)
        fi
    fi
    if [[ -n "$model_dir" && -d "$model_dir" ]]; then
        log_to_file "Detected model directory: $model_dir"
        if command_exists du; then
            log_to_file "#+begin_src sh"
            du -sh "$model_dir" 2>&1 >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
        fi
    else
        note "Could not confidently locate Ollama models directory"
    fi

    subsection "Ollama Environment Variables"
    if command_exists env; then
        log_to_file "#+begin_src sh"
        env | grep -iE 'ollama|OLLAMA' | sort >> "$OUTPUT_FILE" || echo "No OLLAMA_* env vars found" >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    subsection "Ollama Systemd Unit (if managed by systemd)"
    if command_exists systemctl; then
        if systemctl list-unit-files | grep -q ollama; then
            log_to_file "#+begin_src sh"
            systemctl cat ollama.service 2>&1 | head -40 >> "$OUTPUT_FILE" || true
            log_to_file "#+end_src"
            log_to_file "Status: $(systemctl is-active ollama 2>/dev/null || echo 'unknown')"
        else
            note "No ollama systemd unit found (may be running via user service or manually)"
        fi
    fi
}


collect_hermes_and_withnail() {
    section 1 "AI Software Stack - Hermes Agent + Withnail (CEO Bot)"

    subsection "Hermes Agent Detection & Version"
    if command_exists hermes; then
        log_to_file "#+begin_src sh"
        hermes --version 2>&1 >> "$OUTPUT_FILE" || true
        log_to_file "#+end_src"

        subsection "Hermes Doctor / Status (safe run)"
        if command_exists timeout; then
            log_to_file "#+begin_src sh"
            timeout 15s hermes doctor 2>&1 | head -30 >> "$OUTPUT_FILE" || echo "hermes doctor timed out or failed" >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
        else
            note "timeout command not available; skipping hermes doctor"
        fi
    else
        note "hermes command not found in PATH"
    fi

    subsection "Hermes Configuration Files (sanitized)"
    if [[ -d "$KNOWN_HERMES_DIR" ]]; then
        log_to_file "Hermes config directory exists: $KNOWN_HERMES_DIR"
        if [[ -f "${KNOWN_HERMES_DIR}/config.yaml" ]]; then
            subsection "config.yaml (sensitive values redacted)"
            log_to_file "#+begin_src yaml"
            sanitize_yaml "${KNOWN_HERMES_DIR}/config.yaml" >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
        fi
        if [[ -f "${KNOWN_HERMES_DIR}/.env" ]]; then
            log_to_file ""
            log_to_file "NOTE: ~/.hermes/.env exists (contents NOT dumped for security). Review manually if needed."
            ls -l "${KNOWN_HERMES_DIR}/.env" 2>/dev/null >> "$OUTPUT_FILE"
        fi
    else
        note "Standard Hermes directory ${KNOWN_HERMES_DIR} not found"
    fi

    subsection "Hermes Skills / Tools Configuration"
    if command_exists hermes; then
        log_to_file "#+begin_src sh"
        hermes tools 2>&1 | head -20 >> "$OUTPUT_FILE" || echo "hermes tools command not available or produced no output" >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi
    if command_exists find; then
        log_to_file ""
        log_to_file "#+begin_src text"
        echo "Potential skills-related files under ~ (limited depth):" >> "$OUTPUT_FILE"
        find ~ -maxdepth 4 -type f \( -name "*skill*" -o -name "*skills*" -o -name "*tool*" \) 2>/dev/null | head -15 >> "$OUTPUT_FILE" || true
        log_to_file "#+end_src"
    fi

    subsection "Withnail (CEO Bot) Detection"
    local withnail_found="no"
    if command_exists pgrep; then
        if pgrep -i withnail >/dev/null 2>&1; then
            withnail_found="yes (process running)"
            log_to_file "#+begin_src sh"
            pgrep -a -i withnail 2>&1 >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
        fi
    fi

    if command_exists systemctl; then
        if systemctl list-units --type=service | grep -qi withnail; then
            withnail_found="yes (systemd unit)"
            log_to_file "#+begin_src sh"
            systemctl list-units --type=service | grep -i withnail >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
        fi
    fi

    if command_exists find; then
        log_to_file ""
        log_to_file "#+begin_src text"
        echo "Files/directories potentially related to Withnail (limited depth & name match):" >> "$OUTPUT_FILE"
        find ~ /opt -maxdepth 5 \( -iname "*withnail*" -o -iname "*Withnail*" \) 2>/dev/null | head -20 >> "$OUTPUT_FILE" || true
        log_to_file "#+end_src"
    fi

    log_to_file ""
    log_to_file "Withnail detection summary: ${withnail_found}"
    if [[ "$withnail_found" == "yes"* ]]; then
        log_to_file "ACTION: Update top-level :WITHNAIL_DETECTED: property after review."
    fi

    subsection "Telegram Gateway / Integration (high-level only)"
    if command_exists pgrep; then
        if pgrep -iE 'telegram|gateway' >/dev/null 2>&1; then
            log_to_file "Note: Process(es) matching telegram or gateway detected. High-level status only — no tokens shown."
        fi
    fi
}

collect_services_runtime() {
    section 1 "System Services, Automation & Runtime State (Snapshot)"

    subsection "Relevant Systemd Units"
    if command_exists systemctl; then
        log_to_file "#+begin_src sh"
        systemctl list-units --type=service | grep -iE 'ollama|hermes|withnail|gateway|tailscale' | head -20 >> "$OUTPUT_FILE" || echo "No matching units" >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    subsection "User Crontab (filtered)"
    if command_exists crontab; then
        log_to_file "#+begin_src sh"
        crontab -l 2>/dev/null | grep -vE '^#|^$' | head -20 >> "$OUTPUT_FILE" || echo "No user crontab entries" >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    subsection "System Cron Directories (relevant entries)"
    for cronpath in /etc/cron.d /etc/cron.hourly /etc/cron.daily; do
        if [[ -d "$cronpath" ]]; then
            log_to_file "#+begin_src sh"
            ls -1 "$cronpath" 2>/dev/null | head -10 >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
        fi
    done

    subsection "Current Resource Usage Snapshot"
    if command_exists uptime; then
        log_to_file "Load: $(uptime)"
    fi
    if command_exists free; then
        log_to_file "#+begin_src sh"
        free -h >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi
    if command_exists df; then
        log_to_file "#+begin_src sh"
        df -h | head -10 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    subsection "Top Processes (filtered to agent-related)"
    if command_exists top; then
        log_to_file "#+begin_src sh"
        top -b -n1 | head -20 >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi
    if command_exists ps; then
        log_to_file "#+begin_src sh"
        ps auxf | grep -iE 'ollama|hermes|withnail|python' | grep -v grep | head -15 >> "$OUTPUT_FILE" || true
        log_to_file "#+end_src"
    fi

    subsection "Throttling / Power / Thermal History (from earlier vcgencmd)"
    log_to_file "See Hardware section for vcgencmd get_throttled and temperature output."
}

collect_configs() {
    section 1 "Key Configuration Files & Environment"

    subsection "Shell Customizations (~/.bashrc relevant snippets)"
    if [[ -f "${HOME}/.bashrc" ]]; then
        log_to_file "#+begin_src sh"
        grep -iE 'color|prompt|alias|PATH|hermes|ollama|withnail' "${HOME}/.bashrc" 2>/dev/null | head -20 >> "$OUTPUT_FILE" || echo "No relevant lines" >> "$OUTPUT_FILE"
        log_to_file "#+end_src"
    fi

    subsection "Environment Variables (filtered to stack)"
    if command_exists env; then
        log_to_file "#+begin_src sh"
        env | grep -iE 'ollama|hermes|withnail|path|python' | sort | head -30 >> "$OUTPUT_FILE" || true
        log_to_file "#+end_src"
    fi

    subsection "Python Environment (relevant packages)"
    if command_exists python3; then
        log_to_file "Python version: $(python3 --version 2>&1)"
        if command_exists pip3 || command_exists pip; then
            local pipcmd="pip3"
            command_exists pip || pipcmd="pip"
            log_to_file "#+begin_src sh"
            $pipcmd list 2>/dev/null | grep -iE 'ollama|langchain|hermes|torch|transformers|llama|openai|anthropic' | head -20 >> "$OUTPUT_FILE" || echo "No matching Python packages found" >> "$OUTPUT_FILE"
            log_to_file "#+end_src"
        fi
    fi
}

collect_notes() {
    section 1 "Notes / Warnings / Next Steps"

    log_to_file "- This collection was performed in a read-only manner. No system changes were made."
    log_to_file "- Sensitive values in Hermes config files were redacted where patterns were detected. Always manually review the output file before adding to any shared repository."
    log_to_file "- Commands that were unavailable were noted with 'NOTE:' entries. Install missing packages (e.g., raspberrypi-utils for vcgencmd) if deeper hardware data is required."
    log_to_file "- For best long-term value: Re-run this script before and after any significant changes (Hermes/Withnail upgrades, config edits, hardware additions, power supply changes, etc.). Commit the resulting .org files to your knowledge-base Git repository with clear messages."
    log_to_file "- Recommended next steps after retrieval:"
    log_to_file "  1. scp the file(s) to your laptop."
    log_to_file "  2. Open in Emacs, review, and create/update org-roam nodes or Tsuioku entries from major sections."
    log_to_file "  3. Update the top-level :PROPERTIES: drawer with any discovered quick facts (PI_MODEL, RAM_GB, exact versions, etc.)."
    log_to_file "  4. Link this snapshot to related project notes (e.g., Withnail improvements, power optimization, air-gapping checklist)."
    log_to_file ""
    log_to_file "Collection complete at $(date '+%Y-%m-%d %H:%M:%S %Z'). File: ${OUTPUT_FILE}"
}

# -----------------------------------------------------------------------------
# Main Execution
# -----------------------------------------------------------------------------
main() {
    if [[ "${1:-}" == "--help" || "${1:-}" == "-h" ]]; then
        print_usage
        exit 0
    fi

    mkdir -p "$OUTPUT_DIR" || {
        echo "ERROR: Could not create output directory $OUTPUT_DIR" >&2
        exit 1
    }

    : > "$OUTPUT_FILE"

    echo "Starting collection... Output will be written to: $OUTPUT_FILE" >&2

    collect_metadata
    collect_hardware
    collect_os_kernel
    collect_networking
    collect_ollama
    collect_hermes_and_withnail
    collect_services_runtime
    collect_configs
    collect_notes

    ln -sfn "$(basename "$OUTPUT_FILE")" "$LATEST_LINK" 2>/dev/null || true

    echo ""
    echo "==================================================================="
    echo "Collection finished successfully."
    echo "Org-mode file: $OUTPUT_FILE"
    echo "Latest symlink: $LATEST_LINK"
    echo ""
    echo "Next: From your laptop, run something like:"
    echo "  scp gt-hermes@hermes-pi:~/pi-specs/pi-withnail-specs-${DATE_STAMP}.org ./"
    echo "  (or scp ...:~/pi-specs/pi-withnail-specs-latest.org ./)"
    echo ""
    echo "Then open the file in Emacs, review, and ingest into your knowledge base."
    echo "Re-run this script before/after future configuration changes to maintain history."
    echo "==================================================================="
}

main "$@"
